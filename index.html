Workshop 10: Smart Note Hive CRUD

คู่มือเรียนรู้ระบบคลังข้อมูลภายในเครื่องและการควบคุมความปลอดภัยของชนิดข้อมูล (Type Safety)

LEARNING CONTENT

ความรู้พื้นฐานและสถาปัตยกรรมระบบคลังข้อมูล Hive

ในบทเรียนนี้นักศึกษาจะได้เรียนรู้วิธีการบันทึกข้อมูลถาวรภายในเครื่องสมาร์ตโฟน (Local Database) ด้วย Hive Database ซึ่งเป็นคลังข้อมูล NoSQL แบบ Key-Value Store ที่ทำงานได้รวดเร็วและปลอดภัยสูง โดยมีหัวข้อหลักที่ต้องทำความเข้าใจดังนี้:

คุณสมบัติระบบ (Feature) คำอธิบายและการทำงานในแอปพลิเคชัน
Create (คำสั่งบันทึก) ใช้ฟังก์ชัน box.add(data) บันทึกวัตถุใหม่ลงในคลังข้อมูลฐานข้อมูลอัตโนมัติ
Read (คำสั่งดึงค่า) ใช้ฟังก์ชัน box.getAt(index) มาแมปออกผลลัพธ์ผ่าน ListView.builder
Update (คำสั่งแก้ไข) ใช้คำสั่ง box.putAt(index, data) เพื่ออัปเดตข้อมูลทับตำแหน่งดั้งเดิมในฐานข้อมูล
Delete (คำสั่งลบ) ใช้คำสั่ง box.deleteAt(index) เพื่อนำข้อมูลออกจากหน่วยความจำทันที
Real-time Search ระบบกรองคำค้นหา (Filter) แบบไม่แยกแยะอักษรพิมพ์เล็กพิมพ์ใหญ่ (Case-Insensitive)

ให้นักศึกษาสร้างไฟล์ชื่อ note.dart ขึ้นมาใหม่เพื่อทำหน้าที่เป็นโครงสร้างคลาสเก็บข้อมูลบันทึก:

lib/note.dart
class Note {
  String title;
  String content;
  String category;
  String date;

  Note({
    required this.title,
    required this.content,
    required this.category,
    required this.date,
  });

  Map toMap() {
    return {
      'title': title,
      'content': content,
      'category': category,
      'date': date,
    };
  }
}
STEP 4

พัฒนาสถาปัตยกรรมระบบหลัก (lib/main.dart)

ข้อควรระวังสำหรับนักศึกษา: ให้ล้างซอร์สโค้ดในไฟล์ main.dart เดิมออกให้หมด แล้วแทนที่ด้วยชุดรหัสสถาปัตยกรรมระบบที่ได้รับการแก้ไขข้อผิดพลาดเรื่องชนิดข้อมูลและปรับปรุงระบบค้นหาเรียบร้อยแล้วด้านล่างนี้
lib/main.dart
import 'package:flutter/material.dart';
import 'package:hive_flutter/hive_flutter.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Hive.initFlutter();
  await Hive.openBox('notes');
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      theme: ThemeData(primarySwatch: Colors.teal, useMaterial3: true),
      home: const HomePage(),
    );
  }
}

class HomePage extends StatefulWidget {
  const HomePage({super.key});

  @override
  State createState() {
    return _HomePageState();
  }
}

class _HomePageState extends State {
  final titleController = TextEditingController();
  final contentController = TextEditingController();
  final searchController = TextEditingController();

  String category = 'การเรียน';
  int editIndex = -1;

  final List categoryList = [
    'การเรียน',
    'งานวิจัย',
    'ประชุม',
    'ส่วนตัว',
  ];

  Future saveNote() async {
    if (titleController.text.isEmpty || contentController.text.isEmpty) return;

    final box = Hive.box('notes');
    Map data = {
      'title': titleController.text,
      'content': contentController.text,
      'category': category,
      'date': DateTime.now().toString().substring(0, 16),
    };

    if (editIndex == -1) {
      await box.add(data);
    } else {
      await box.putAt(editIndex, data);
      editIndex = -1;
    }

    titleController.clear();
    contentController.clear();
    category = 'การเรียน';
    setState(() {});
  }

  @override
  Widget build(BuildContext context) {
    final box = Hive.box('notes');

    return Scaffold(
      appBar: AppBar(
        title: const Text('Smart Note Hive'),
        backgroundColor: Colors.teal,
        foregroundColor: Colors.white,
      ),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            TextField(
              controller: searchController,
              decoration: InputDecoration(
                labelText: 'Search',
                prefixIcon: const Icon(Icons.search),
                border: const OutlineInputBorder(),
                suffixIcon: searchController.text.isNotEmpty
                    ? IconButton(
                        icon: const Icon(Icons.clear),
                        onPressed: () {
                          searchController.clear();
                          setState(() {});
                        },
                      )
                    : null,
              ),
              onChanged: (v) {
                setState(() {});
              },
            ),
            const SizedBox(height: 12),
            TextField(
              controller: titleController,
              decoration: const InputDecoration(
                labelText: 'Title',
                border: OutlineInputBorder(),
              ),
            ),
            const SizedBox(height: 12),
            TextField(
              controller: contentController,
              decoration: const InputDecoration(
                labelText: 'Content',
                border: OutlineInputBorder(),
              ),
            ),
            const SizedBox(height: 12),
            // 🎯 จุดที่แก้ไข: กำหนด Type ชัดเจน ป้องกัน Error แดงตรง .map() และ .toList()
            DropdownButtonFormField(
              decoration: const InputDecoration(
                labelText: 'Category',
                border: OutlineInputBorder(),
              ),
              isExpanded: true,
              value: category,
              items: categoryList.map>((String e) {
                return DropdownMenuItem(value: e, child: Text(e));
              }).toList(),
              onChanged: (String? v) {
                if (v == null) return;
                setState(() {
                  category = v;
                });
              },
            ),
            const SizedBox(height: 12),
            SizedBox(
              width: double.infinity,
              child: ElevatedButton(
                onPressed: saveNote,
                style: ElevatedButton.styleFrom(
                  backgroundColor: Colors.teal,
                  foregroundColor: Colors.white,
                ),
                child: Text(editIndex == -1 ? 'Save Note' : 'Update Note'),
              ),
            ),
            const SizedBox(height: 20),
            Expanded(
              child: ValueListenableBuilder(
                valueListenable: box.listenable(),
                builder: (context, Box cleanBox, _) {
                  return ListView.builder(
                    itemCount: cleanBox.length,
                    itemBuilder: (context, index) {
                      final data = cleanBox.getAt(index);

                      if (searchController.text.isNotEmpty &&
                          !data['title'].toString().toLowerCase().contains(
                            searchController.text.toLowerCase(),
                          )) {
                        return Container();
                      }

                      return Card(
                        child: ListTile(
                          title: Text(
                            data['title'],
                            style: const TextStyle(fontWeight: FontWeight.bold),
                          ),
                          subtitle: Text(
                            "${data['content']}\n${data['category']} | ${data['date']}",
                          ),
                          leading: IconButton(
                            icon: const Icon(Icons.edit, color: Colors.blue),
                            onPressed: () {
                              titleController.text = data['title'];
                              contentController.text = data['content'];
                              category = data['category'];
                              editIndex = index;
                              setState(() {});
                            },
                          ),
                          trailing: IconButton(
                            icon: const Icon(Icons.delete, color: Colors.red),
                            onPressed: () async {
                              await cleanBox.deleteAt(index);
                            },
                          ),
                        ),
                      );
                    },
                  );
                },
              ),
            ),
          ],
        ),
      ),
    );
  }
}
ระบบงานนี้ทำงานได้อย่างสมบูรณ์แบบ รองรับฟังก์ชันมิติข้อมูลครบองค์ประกอบ CRUD, ป้องกันข้อผิดพลาดเรื่องความสอดคล้องชนิดข้อมูล และกรองข้อมูลแบบอัจฉริยะเรียบร้อยแล้ว
