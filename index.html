lib/note.dart 
class Note {

  final String title;
  final String content;

  Note({
    required this.title,
    required this.content,
  });

  Map toMap(){

    return {

      'title':title,
      'content':content,

    };

  }

  factory Note.fromMap(
    Map map
  ){

    return Note(

      title: map['title'],
      content: map['content'],

    );

  }

}
✅ STEP นี้ต้องได้
  • มีไฟล์ note.dart

  • ยังไม่ต้อง Run
STEP 3

CREATE — เพิ่ม Note

STEP นี้แก้เฉพาะ main.dart

lib/ ├── main.dart ← แก้ไฟล์นี้ └── note.dart
สำคัญมาก ฟังก์ชัน addNote() ต้องอยู่ภายใน class _HomePageState เท่านั้น

เปิด main.dart

ลบทุกอย่างในไฟล์

วางโค้ดนี้ทั้งหมด

lib/main.dart 
import 'package:flutter/material.dart';
import 'package:hive_flutter/hive_flutter.dart';

import 'note.dart';

void main() async {

  WidgetsFlutterBinding.ensureInitialized();

  await Hive.initFlutter();

  await Hive.openBox('notes');

  runApp(
    const MyApp()
  );

}

class MyApp extends StatelessWidget {

  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {

    return MaterialApp(

      debugShowCheckedModeBanner:false,

      home: HomePage(),

    );

  }

}

class HomePage extends StatefulWidget {

  const HomePage({super.key});

  @override
  State createState() {

    return _HomePageState();

  }

}

class _HomePageState extends State {

  final titleController =
    TextEditingController();

  final contentController =
    TextEditingController();

  Future addNote() async {

    final box =
      Hive.box('notes');

    final note = Note(

      title:
        titleController.text,

      content:
        contentController.text,

    );

    await box.add(
      note.toMap()
    );

    titleController.clear();

    contentController.clear();

  }

  @override
  Widget build(BuildContext context) {

    return Scaffold(

      appBar: AppBar(

        title:
          const Text('Note Keeper'),

      ),

      body: Padding(

        padding:
          const EdgeInsets.all(16),

        child: Column(

          children: [

            TextField(

              controller:
                titleController,

              decoration:
                const InputDecoration(

                  labelText:'Title',

                ),

            ),

            const SizedBox(
              height:12,
            ),

            TextField(

              controller:
                contentController,

              decoration:
                const InputDecoration(

                  labelText:'Content',

                ),

            ),

            const SizedBox(
              height:16,
            ),

            ElevatedButton(

              onPressed:addNote,

              child:
                const Text(
                  'Save Note'
                ),

            ),

          ],

        ),

      ),

    );

  }

}
✅ STEP นี้ต้องได้
  • ไม่มีเส้นแดง
  • กด Run ได้
  • เห็น TextField 2 ช่อง
  • เห็นปุ่ม Save Note
STEP 4

READ — แสดงรายการ

STEP นี้เพิ่ม ListView เพื่อแสดง Note ที่บันทึกไว้

แก้เฉพาะใน build() ไม่ต้องลบทั้งไฟล์

หาโค้ดนี้

main.dart 
ElevatedButton(

  onPressed:addNote,

  child:
    const Text(
      'Save Note'
    ),

),

วางโค้ดด้านล่างต่อจาก ElevatedButton

เพิ่มด้านล่างปุ่ม 
const SizedBox(
  height:20,
),

Expanded(

  child: ValueListenableBuilder(

    valueListenable:
      Hive.box('notes')
      .listenable(),

    builder:
      (context, Box box, _) {

      return ListView.builder(

        itemCount:
          box.length,

        itemBuilder:
          (context,index){

          final map =
            box.getAt(index);

          final note =
            Note.fromMap(map);

          return Card(

            child: ListTile(

              title:
                Text(note.title),

              subtitle:
                Text(note.content),

            ),

          );

        },

      );

    },

  ),

)
✅ STEP นี้ต้องได้
  • กด Save แล้วข้อมูลขึ้นทันที
  • ไม่ crash
  • ListView แสดงข้อมูลได้
